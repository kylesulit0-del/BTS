**Active state (iframe playing):**
Same two-zone structure:
```tsx
return (
  <div className="snap-card-video">
    <div className="snap-card-media-zone">
      {/* Loading state (thumbnail + spinner) -- inside media zone */}
      {!iframeLoaded && (
        <div className="snap-card-video-facade snap-card-video-loading">
          {/* thumbnail or dark gradient -- same as before but positioned within media zone */}
        </div>
      )}

      {/* Iframe -- inside media zone */}
      <iframe ... style={{
        position: 'absolute',
        inset: 0,
        width: '100%',
        height: '100%',
        border: 'none',
        opacity: iframeLoaded ? 1 : 0,
        zIndex: 1,
      }} />

      {/* Touch overlay -- inside media zone, will auto-constrain to zone via position:relative parent */}
      <div className="snap-video-touch-overlay" ... />

      {/* Tap icon -- inside media zone */}
      {tapIcon && <div className="snap-video-tap-icon" key={tapIcon}>...</div>}

      {/* Mute button -- inside media zone */}
      <button className="snap-card-video-mute-btn" ...>...</button>

      {/* Progress bar -- inside media zone */}
      {videoType === 'youtube-short' && iframeLoaded && (
        <div className="snap-video-progress">...</div>
      )}
    </div>

    {/* Info panel -- below media zone */}
    <InfoPanel item={item} />
  </div>
);
```
